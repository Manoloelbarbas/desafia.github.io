/* 1. ANTES: PROCESO MANUAL */
async function procesarFacturas(pdfs) {
  let resultados = [];
  for (let archivo of pdfs) {
    const texto = await ocr.leer(archivo);
    const datos = regex.extraer(texto);
    
    if (!datos.valido) {
      throw new Error("Se necesita revisión manual");
    }
    resultados.push(datos);
  }
  return bd.guardar(resultados);
}
